## Dark Mode with next-themes

### Setup
```bash
npm install next-themes
```

```tsx
// app/layout.tsx
import { ThemeProvider } from "next-themes"

export default function RootLayout({ children }) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        <ThemeProvider attribute="class" defaultTheme="system" enableSystem disableTransitionOnChange>
          {children}
        </ThemeProvider>
      </body>
    </html>
  )
}
```

### Theme Toggle
```tsx
"use client"
import { Moon, Sun } from "lucide-react"
import { useTheme } from "next-themes"
import { Button } from "@/components/ui/button"
import { DropdownMenu, DropdownMenuContent, DropdownMenuItem, DropdownMenuTrigger } from "@/components/ui/dropdown-menu"

export function ThemeToggle() {
  const { setTheme } = useTheme()
  return (
    <DropdownMenu>
      <DropdownMenuTrigger asChild>
        <Button variant="outline" size="icon">
          <Sun className="h-4 w-4 rotate-0 scale-100 transition-all dark:-rotate-90 dark:scale-0" />
          <Moon className="absolute h-4 w-4 rotate-90 scale-0 transition-all dark:rotate-0 dark:scale-100" />
        </Button>
      </DropdownMenuTrigger>
      <DropdownMenuContent>
        <DropdownMenuItem onClick={() => setTheme("light")}>Light</DropdownMenuItem>
        <DropdownMenuItem onClick={() => setTheme("dark")}>Dark</DropdownMenuItem>
        <DropdownMenuItem onClick={() => setTheme("system")}>System</DropdownMenuItem>
      </DropdownMenuContent>
    </DropdownMenu>
  )
}
```

## Custom Color Palettes

### Brand Theme Example
```css
:root {
  /* Custom brand colors */
  --brand: 262 83% 58%;          /* Purple */
  --brand-foreground: 0 0% 100%;
  --accent: 173 58% 39%;          /* Teal */
  --success: 142 76% 36%;         /* Green */
  --warning: 38 92% 50%;          /* Amber */
}
```

### Using in Components
```tsx
<div className="bg-primary text-primary-foreground">Uses theme colors</div>
<div className="bg-muted text-muted-foreground">Muted section</div>
<div className="border-border">Themed border</div>
<div className="text-destructive">Error text</div>

/* Custom with CSS */
<div style={{ background: "hsl(var(--brand))" }}>Brand color</div>
```

## Tailwind v4 Theming

```css
/* Tailwind v4: Use @theme directive */
@theme {
  --color-brand: oklch(0.65 0.24 285);
  --color-brand-light: oklch(0.85 0.12 285);
  --font-display: "Inter", sans-serif;
  --breakpoint-3xl: 1920px;
}
```

Usage: `<div className="text-brand font-display 3xl:text-xl">...</div>`
